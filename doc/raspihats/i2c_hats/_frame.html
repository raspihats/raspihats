 45class Frame(object):
 46    """The Frame is used for communication over the I2C bus:
 47
 48    +----+---------+------+------------------------------------------------------------------------------------------+
 49    | #  | Field   | Size | Description                                                                              |
 50    +====+=========+======+==========================================================================================+
 51    | 1. | Id      | 1    | Diferent from one request to another, I2C-HAT responds with the same Id byte.            |
 52    +----+---------+------+------------------------------------------------------------------------------------------+
 53    | 2. | Command | 1    | Defines the action to be taken by the I2C-HAT which responds with the same Command byte. |
 54    +----+---------+------+------------------------------------------------------------------------------------------+
 55    | 3. | Data    | n    | Payload data.                                                                            |
 56    +----+---------+------+------------------------------------------------------------------------------------------+
 57    | 4. | Crc     | 2    | Modbus CRC16 for data integrity.                                                         |
 58    +----+---------+------+------------------------------------------------------------------------------------------+
 59
 60    Args:
 61        id (:obj:`int`): ID byte
 62        cmd (:obj:`int`): Command byte
 63        data (:obj:`list` of :obj:`int`): Payload data bytes
 64
 65    Attributes:
 66        id (:obj:`int`): ID byte
 67        cmd (:obj:`int`): Command byte
 68        data (:obj:`list` of :obj:`int`): Payload data bytes
 69
 70    """
 71
 72    # byte size for fields
 73    ID_SIZE = 1
 74    CMD_SIZE = 1
 75    CRC_SIZE = 2
 76
 77    def __init__(self, id, cmd, data=[]):
 78        self.id = id
 79        self.cmd = Command(cmd)
 80        self.data = data
 81
 82    def encode(self):
 83        """Encode the frame fields: Id, Command, Data and Crc to a list of ints.
 84
 85        Returns:
 86            :obj:`list` of :obj:`int`: List of frame bytes, raw data that can be transmitted over the I2C bus
 87
 88        """
 89        data = [self.id, self.cmd.value] + self.data
 90        crc = crc16.modbus(data)
 91        return data + [(crc & 0xFF), ((crc >> 8) & 0xFF)]
 92
 93    def decode(self, data):
 94        """Decode raw data from I2C bus. It's used to decode the I2C-HATs response. The fields Id and Command should already be set
 95        because a valid I2C-HAT response always has the same Id and Command bytes as the request.
 96
 97        Args:
 98            data (:obj:`list` of :obj:`int`): Raw I2C data to be decoded
 99
100        Raises:
101            :obj:`DecodeException`: If the response frame Crc check fails, or has an unexpected Id or Command
102
103        """
104        crc = crc16.modbus(data[:-2])
105        crc_in = (data[-1] << 8) + data[-2]
106        if crc != crc_in:
107            #print('crc check failed, ' + hex(crc) + '!=' + hex(crc_in) + str([hex(x) for x in data]))
108            raise DecodeException('crc check failed, ' + hex(crc) + '!=' + hex(crc_in) + ' data:' + str([hex(x) for x in data]))
109        if self.id != data[0]:
110            #print('unexpected id')
111            raise DecodeException('unexpected id')
112        if self.cmd.value != data[1]:
113            #print('unexpected command')
114            raise DecodeException('unexpected command')
115        self.data = data[2:-2]
