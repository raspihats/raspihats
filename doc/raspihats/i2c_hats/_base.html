 37    def __init__(self, address, base_address=None, board_name=None):
 38
 39        if I2CHat._i2c_bus is None:
 40            I2CHat._i2c_bus = smbus2.SMBus(I2CHat.I2C_PORT)
 41
 42        self._address = address
 43        self._frame_id = 0
 44        self._transfer_time = None
 45
 46        if base_address == None:
 47            if not 0 <= address <= 127:
 48                raise ValueError("I2C address should be in range[0, 127]")
 49        else:
 50            if address & base_address != base_address:
 51                raise ValueError("I2C address should be in range[" + hex(base_address) + ", " + hex(base_address + 0x0F) + "]")
 52
 53        if board_name != None:
 54            if self.name not in board_name:
 55                raise Exception("unexpected board name '" + self.name + "', expecting '" + board_name + "'")

 63        return string

 65    def _generate_frame_id_(self):
 66        """Generate new frame Id, increments current frame Id, wraps to 0x7F beacuse of Raspberry Pi I2C bug which affects MSb
 67
 68        Returns:
 69            int: The new frame Id
 70
 71        """
 72        self._frame_id += 1
 73        self._frame_id &= 0x7F
 74        return self._frame_id

 76    def _transfer_(self, request_frame, response_data_size, response_expected=True, number_of_tries=5):
 77        """Tries a number of times to send a request frame and to get a response frame over I2C bus.
 78
 79        Args:
 80            request_frame (Frame): Request frame to be sent over the I2C bus
 81            response_data_size (int): Expected response data size, this is the payload data size
 82            response_expected (bool): True if a respose is expected
 83            number_of_tries (int): Number of tries to get the response
 84
 85        Returns:
 86            Frame: The response frame
 87
 88        Raises:
 89            ResponseException: After all attempts to get a response have failed
 90
 91        """
 92        with I2CHat._i2c_bus_lock:
 93            try_cnt = 1
 94            while True:
 95                try:
 96                    request_data = request_frame.encode()
 97
 98                    # NOTE: write_i2c_block_data function is used to send commands to the I2C-HAT
 99                    I2CHat._i2c_bus.write_i2c_block_data(self._address, request_data[0], request_data[1:])
100
101                    if not response_expected:
102                        return
103
104                    # NOTE: read_i2c_block_data function sends a i2c_write first, this write has a length of one, and the dummy_byte as payload, this
105                    # write will be ignored by the I2C-HAT, after this a i2c_read will be issued, this i2c_read is used for reading the response
106                    dummy_byte = 0xFF
107                    expected_response_size = Frame.ID_SIZE + Frame.CMD_SIZE + response_data_size + Frame.CRC_SIZE
108                    response_data = I2CHat._i2c_bus.read_i2c_block_data(self._address, dummy_byte, expected_response_size)
109
110                    # build response frame
111                    response_frame = Frame(request_frame.id, request_frame.cmd)
112                    response_frame.decode(response_data)
113                    self._transfer_time = time.time()
114                    return response_frame
115
116                except (IOError, DecodeException) as ex:
117                    if try_cnt >= number_of_tries:
118                        if isinstance(ex, IOError):
119                            raise ResponseException("no response")
120                        else:
121                            raise ResponseException(str(ex))
122                    time.sleep(0.01)
123                    try_cnt += 1

125    def _get_u32_value_(self, cmd):
126        """Generic get for a unsigned32 value.
127
128        Args:
129            cmd (int): Command byte value
130
131        Returns:
132            int: The desired unsigned32 value
133
134        Raises:
135            ResponseException: If response has bad data length
136
137        """
138        request = self._request_frame_(cmd, [])
139        response = self._transfer_(request, 4)
140        data = response.data
141        if len(data) != 4:
142            raise ResponseException('invalid response data length')
143        return data[0] + (data[1] << 8) + (data[2] << 16) + (data[3] << 24)

154

156            ResponseException: I response

158        data = [value & 0xFF, (value >> 8) & 0xFF, (value >> 16) & 0xFF, (value >> 24) & 0xFF]

161        if data != response.data:

251class Functionality(object):
252    """I2C-HAT functionality base.
253
254    Args:
255        i2c_hat (:obj:`raspihats.i2c_hats._base.I2CHat`): I2CHat instance
256        labels (:obj:`list` of :obj:`str` or optional): Channel labels
257
258    """
259
260    def __init__(self, i2c_hat, labels=None):
261        self._i2c_hat = i2c_hat
262        self._labels = labels
263        if labels != None:
264            self._lc_labels = [l.lower() for l in labels]
265
266    def _validate_channel_index(self, index):
267        if self._labels == None:
268            raise Exception('no labels defined')
269
270        label = None
271        if isinstance(index, int):
272            if not (0 <= index < len(self._labels)):
273                raise IndexError("'" + str(index) + "' is not a valid channel index")
274        elif isinstance(index, str):
275            label = index
276            try:
277                index = self._lc_labels.index(label.lower())
278            except ValueError:
279                raise ValueError("'" + label + "' is not a valid channel label")
280        else:
281            raise ValueError("index type is '" + type(index) + "', expecting 'int' or 'str'")
282        return index
283
284    def _validate_value(self, value):
285        max_value = (0x01 << len(self._labels)) - 1
286        if not (0 <= value <= max_value):
287            raise ValueError("'" + str(value) + "' is not a valid value, range is [0x00 .. " + hex(max_value) + "]")
288
289    @property
290    def labels(self):
291        """:obj:`list` of :obj:`str`: Channel Labels."""
292        return self._labels

338    def get_reg(self, reg_type):
339        """:obj:`int`: The value of IRQ control reg, 1 bit represents 1 channel."""
340        request = self._i2c_hat._request_frame_(Command.IRQ_GET_REG, [reg_type])
341        response = self._i2c_hat._transfer_(request, 5)
342        data = response.data
343        if (len(data) != 1 + 4) or (data[0] != reg_type):
344            raise ResponseException('Invalid data')
345        return data[1] + (data[2] << 8) + (data[3] << 16) + (data[4] << 24)
346
347    def set_reg(self, reg_type, value):
348        """:obj:`int`: The value of IRQ control reg, 1 bit represents 1 channel."""
349        # self._validate_value(value)
350        data = [reg_type, value & 0xFF, (value >> 8) & 0xFF, (value >> 16) & 0xFF, (value >> 24) & 0xFF]
351        request = self._i2c_hat._request_frame_(Command.IRQ_SET_REG, data)
352        response = self._i2c_hat._transfer_(request, 5)
353        data = response.data
354        if (len(data) != 1 + 4) or (data[0] != reg_type):
